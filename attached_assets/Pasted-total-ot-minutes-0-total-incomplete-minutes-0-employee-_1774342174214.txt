total_ot_minutes = 0
total_incomplete_minutes = 0

employee = frappe.get_doc("Employee", doc.employee)

designation = employee.designation
department = employee.department


attendance_days = frappe.db.sql("""
    SELECT attendance_date
    FROM `tabAttendance`
    WHERE employee=%s
    AND attendance_date BETWEEN %s AND %s
    AND docstatus=1
""",(doc.employee,doc.start_date,doc.end_date),as_dict=1)



for day in attendance_days:

    date = day.attendance_date


    # SECURITY NIGHT SHIFT
    if department == "Security - D":

        punches = frappe.db.sql("""
            SELECT time
            FROM `tabEmployee Checkin`
            WHERE employee=%s
            AND time BETWEEN %s
            AND DATE_ADD(%s, INTERVAL 1 DAY)
            ORDER BY time
        """,(doc.employee,date,date),as_dict=1)

    else:

        punches = frappe.db.sql("""
            SELECT time
            FROM `tabEmployee Checkin`
            WHERE employee=%s
            AND DATE(time)=%s
            ORDER BY time
        """,(doc.employee,date),as_dict=1)



    if not punches or len(punches) < 2:
        continue



    # REMOVE DUPLICATE PUNCHES (within 5 minutes)
    filtered_punches = []

    last_time = None

    for p in punches:

        if not last_time:
            filtered_punches.append(p.time)
            last_time = p.time

        else:

            diff = (p.time - last_time).total_seconds() / 60

            if diff >= 5:
                filtered_punches.append(p.time)
                last_time = p.time



    if len(filtered_punches) < 2:
        continue



    # CALCULATE WORK TIME USING PAIRS
    work_minutes = 0

    i = 0

    while i < len(filtered_punches)-1:

        in_time = filtered_punches[i]
        out_time = filtered_punches[i+1]

        diff = (out_time - in_time).total_seconds() / 60

        if diff > 0:
            work_minutes += diff

        i += 2



    # LUNCH RULES
    weekday = date.strftime("%A")

    lunch_minutes = 0


    if department == "Security - D":

        lunch_minutes = 0


    elif department == "Kitchen - D":

        lunch_start = filtered_punches[0].replace(hour=14, minute=0, second=0)

        if filtered_punches[-1] > lunch_start:

            if weekday in ["Monday","Tuesday","Thursday","Friday"]:
                lunch_minutes = 120

            elif weekday in ["Wednesday","Sunday"]:
                lunch_minutes = 60


    else:

        lunch_start = filtered_punches[0].replace(hour=13, minute=0, second=0)

        if filtered_punches[-1] > lunch_start:
            lunch_minutes = 60



    work_minutes = max(work_minutes - lunch_minutes, 0)



    required_minutes = 540
    ot_start = required_minutes + 30



    # OT CALCULATION
    if work_minutes > ot_start:

        if designation != "Manager":

            ot_minutes = work_minutes - ot_start
            total_ot_minutes += ot_minutes



    # INCOMPLETE HOURS
    elif work_minutes < required_minutes:

        if department != "Surf Instructors - D":

            missing_minutes = required_minutes - work_minutes
            total_incomplete_minutes += missing_minutes



doc.custom_ot_hours = round(total_ot_minutes / 60, 2)
doc.custom_incomplete_hours = round(total_incomplete_minutes / 60, 2)



# GET BASIC SALARY
basic_salary = 0

for row in doc.earnings:

    if row.salary_component == "Basic":

        basic_salary = row.amount or 0
        break



daily_rate = basic_salary / 30
hour_rate = daily_rate / 8
minute_rate = hour_rate / 60



ot_amount = total_ot_minutes * minute_rate
incomplete_amount = total_incomplete_minutes * minute_rate



# CLEAN OLD ROWS

earn_rows = []

for e in doc.earnings:

    if e.salary_component != "OT Amount":
        earn_rows.append(e)

doc.earnings = earn_rows



ded_rows = []

for d in doc.deductions:

    if d.salary_component != "Incomplete Working Hours":
        ded_rows.append(d)

doc.deductions = ded_rows



# ADD OT EARNING

if ot_amount > 0:

    doc.append("earnings",{
        "salary_component":"OT Amount",
        "amount":ot_amount
    })



# ADD INCOMPLETE DEDUCTION

if incomplete_amount > 0:

    doc.append("deductions",{
        "salary_component":"Incomplete Working Hours",
        "amount":incomplete_amount
    })
